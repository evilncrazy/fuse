Fuse

Fuse is a C++11 library for manipulating lists of types and expression templates.

Examples

#include "fuse/fuse.hpp"

template <class...Args>
void variadic_func() {
   // Checks if all the variadic arguments are const
   static_assert(
      fuse::all<std::is_const, fuse::make_seq<Args...>::type>::value,
      "All args must be const");

   // Remove all the const qualifiers
   typedef fuse::map<std::remove_const,
      fuse::make_seq<Args...>::type>::type no_const;

   // Now find the common type between them
   typedef fuse::reduce<std::common_type, no_const>::type common_type;
}
